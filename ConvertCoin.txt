<?php
class ConversorMoneda {
  private $tasaCambio;

  public function __construct($tasaCambio) {
    $this->tasaCambio = $tasaCambio;
  }

  public function convertir($montoEnOrigen, $monedaOrigen, $monedaDestino) {
    if ($monedaOrigen == "Bolvares" && $monedaDestino == "Dlares") {
      return $montoEnOrigen / $this->tasaCambio;
    } elseif ($monedaOrigen == "Dlares" && $monedaDestino == "Bolvares") {
      return $montoEnOrigen * $this->tasaCambio;
    } else {
      return "No se puede convertir entre esas monedas";
    }
  }
}

// Ejemplo de uso
$conversor = new ConversorMoneda(1);
echo $conversor->convertir(34.85, "Bolvares", "Dlares"); // Imprime: 1
echo $conversor->convertir(1, "Dlares", "Bolvares"); // Imprime: 34.85
?>
